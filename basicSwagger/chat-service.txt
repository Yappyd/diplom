openapi: 3.0.3

info:
  title: Chat Service API
  version: 1.0.0
  description: |
    Сервис управления чатами.

    Chat Service отвечает за:
      - создание приватных и групповых чатов;
      - хранение структуры чатов;
      - хранение состава групповых чатов;
      - управление ролями участников групповых чатов;
      - проверку принадлежности пользователя к чату.

    Важно:
      - Chat Service не хранит сообщения;
      - Chat Service не хранит профили пользователей;
      - для приватных чатов участники хранятся в таблице private_chats;
      - для групповых чатов участники хранятся в таблице chat_participants;
      - текущий пользователь всегда определяется по JWT, а не передаётся в теле запроса.

        Текущий пользователь определяется по JWT.

        с которым нужно открыть приватный чат.

          description: Идентификатор существующего или нового приватного чата
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/CreatePrivateChatResponse'

          description: Некорректный запрос, например попытка создать чат с самим собой

  /chats/{chatId}/participants/{userId}:
    delete:
      tags:
        - Participants
      summary: Удалить участника из группового чата или выйти из него
      description: |
        Удаляет участника из группового чата.

        Endpoint предназначен только для GROUP-чатов.
        Из PRIVATE-чата выйти нельзя.

        Если userId совпадает с текущим пользователем,
        операция означает выход из группового чата.

        Если userId не совпадает с текущим пользователем,
        операция означает удаление другого участника.

        Требуемые права:
          - пользователь может удалить сам себя;
          - ADMIN может удалить MEMBER;
          - OWNER может удалить ADMIN и MEMBER;
          - OWNER не должен быть удалён, если он последний OWNER в чате.
      parameters:
        - $ref: '#/components/parameters/ChatId'
        - $ref: '#/components/parameters/UserId'
      responses:
        '204':
          description: Участник удалён
        '400':
          description: Некорректный запрос или операция недоступна для PRIVATE-чата
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
        '401':
          description: Пользователь не аутентифицирован
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
        '403':
          description: Недостаточно прав
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
        '404':
          description: Чат или участник не найден
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'
        '409':
          description: Операция нарушает правила чата, например нельзя удалить последнего OWNER
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorResponse'

components:

    bearerAuth:
      type: http
      scheme: bearer
      bearerFormat: JWT

      name: chatId
      in: path
      required: true
      description: Идентификатор чата
      schema:
        type: string
        format: uuid

    UserId:
      name: userId
      in: path
      required: true
      description: Идентификатор пользователя
      schema:
        type: string
        format: uuid

  schemas:

    ChatType:
      type: string
      description: Тип чата
      enum:
        - PRIVATE
        - GROUP
      example: PRIVATE

    ParticipantRole:
      type: string
      description: Роль пользователя в групповом чате
      enum:
        - OWNER
        - ADMIN
        - MEMBER
      example: MEMBER

    CreatePrivateChatRequest:
      type: object
      description: Запрос на создание или получение приватного чата
      required:
        - targetUserId
      properties:
        targetUserId:
          type: string
          format: uuid
          description: Пользователь, с которым нужно создать или получить приватный чат
          example: 11111111-1111-1111-1111-111111111111

    CreatePrivateChatResponse:
      type: object
      description: Ответ на создание или получение приватного чата
      required:
        - chatId
      properties:
        chatId:
          type: string
          format: uuid
          description: Идентификатор существующего или нового приватного чата
          example: 22222222-2222-2222-2222-222222222222

    CreateGroupChatRequest:
      type: object
      description: Запрос на создание группового чата
      required:
        - title
      properties:
        title:
          type: string
          description: Название группового чата
          minLength: 1
          maxLength: 255
          example: Команда проекта
        participantIds:
          type: array
          description: |
            Пользователи, которых нужно добавить в групповой чат как MEMBER.
            Текущего пользователя указывать не нужно — он будет добавлен автоматически как OWNER.
          uniqueItems: true
          items:
            type: string
            format: uuid
          example:
            - 11111111-1111-1111-1111-111111111111
            - 22222222-2222-2222-2222-222222222222

    UpdateChatRequest:
      type: object
      description: Запрос на обновление группового чата
      properties:
        title:
          type: string
          description: Новое название группового чата
          minLength: 1
          maxLength: 255
          example: Новое название группы

    AddParticipantRequest:
      type: object
      description: Запрос на добавление участника в групповой чат
      required:
        - userId
      properties:
        userId:
          type: string
          format: uuid
          description: Идентификатор пользователя, которого нужно добавить
          example: 11111111-1111-1111-1111-111111111111

    ChangeRoleRequest:
      type: object
      description: Запрос на изменение роли участника группового чата
      required:
        - role
      properties:
        role:
          $ref: '#/components/schemas/ParticipantRole'

    ChatResponse:
      type: object
      description: Информация о чате
      required:
        - chatId
        - type
        - createdBy
        - createdAt
        - updatedAt
      properties:
        chatId:
          type: string
          format: uuid
          description: Идентификатор чата
          example: 22222222-2222-2222-2222-222222222222
        type:
          $ref: '#/components/schemas/ChatType'
        title:
          type: string
          nullable: true
          description: |
            Название чата.
            Для GROUP-чата обязательно.
            Для PRIVATE-чата обычно null.
          example: Команда проекта
        createdBy:
          type: string
          format: uuid
          description: Пользователь, создавший чат
          example: 33333333-3333-3333-3333-333333333333

          type: string
          format: date-time
          description: Дата и время создания чата
          example: 2026-04-26T12:00:00Z
        updatedAt:
          type: string
          format: date-time
          description: Дата и время последнего обновления чата
          example: 2026-04-26T12:30:00Z
        privateChat:
          allOf:
            - $ref: '#/components/schemas/PrivateChatInfo'
          nullable: true
          description: |
            Данные приватного чата.
            Заполняется только для PRIVATE-чата.

    PrivateChatInfo:
      type: object
      description: Информация о паре пользователей в приватном чате
      required:
        - userA
        - userB
      properties:
        userA:
          type: string
          format: uuid
          description: Первый пользователь нормализованной пары
          example: 11111111-1111-1111-1111-111111111111
        userB:
          type: string
          format: uuid
          description: Второй пользователь нормализованной пары
          example: 22222222-2222-2222-2222-222222222222

    ChatListResponse:
      type: object
      description: Список чатов текущего пользователя
      required:
        - chats
        - page
        - size
        - totalElements
        - totalPages
      properties:
        chats:
          type: array
          items:
            $ref: '#/components/schemas/ChatResponse'
        page:
          type: integer
          description: Номер текущей страницы
          example: 0
        size:
          type: integer
          description: Размер страницы
          example: 20
        totalElements:
          type: integer
          format: int64
          description: Общее количество элементов
          example: 42
        totalPages:
          type: integer
          description: Общее количество страниц
          example: 3

    ParticipantResponse:
      type: object
      description: Участник группового чата
      required:
        - userId
        - role
        - joinedAt
        - updatedAt
      properties:
        userId:
          type: string
          format: uuid
          description: Идентификатор пользователя
          example: 11111111-1111-1111-1111-111111111111
        role:
          $ref: '#/components/schemas/ParticipantRole'
        nickname:
          type: string
          nullable: true
          description: Локальный никнейм пользователя в групповом чате
          example: Backend Lead
        joinedAt:
          type: string
          format: date-time
          description: Дата и время добавления в групповой чат
          example: 2026-04-26T12:00:00Z
        updatedAt:
          type: string
          format: date-time
          description: Дата и время последнего обновления участника
          example: 2026-04-26T12:30:00Z

    ParticipantsResponse:
      type: object
      description: Список участников группового чата
      required:
        - participants
      properties:
        participants:
          type: array
          items:
            $ref: '#/components/schemas/ParticipantResponse'

        message:
          type: string

        path:
          type: string
          description: Путь запроса
          example: /chats/private
        timestamp:
          type: string
